# Команди для деплою виправлення синхронізації коментарів з HubSpot

# 1. Підключитися до сервера
ssh pro-part-agent@agent.pro-part.online

# 2. Перейти в директорію проекту
cd /home/pro-part-agent/htdocs/agent.pro-part.online/propart-real-estate-hub

# 3. Оновити код з Git
git pull origin main

# 4. Перезапустити додаток
# Якщо використовується systemd:
sudo systemctl restart propart

# Або якщо запущено через nohup:
pkill -f "python.*run.py"
sleep 2
nohup venv/bin/python run.py > logs/propart.log 2>&1 &

# 5. Перезапустити Nginx (якщо потрібно)
sudo systemctl reload nginx

# 6. Перевірити логи після додавання коментаря
tail -f logs/propart.log | grep -i "comment\|hubspot\|note"

# Або переглянути всі логи
tail -f logs/propart.log
